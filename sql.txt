CREATE TABLE tblLocations (
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    roomId VARCHAR(30) NOT NULL,
    locationName VARCHAR(255) NOT NULL,
    locationAddress VARCHAR(255) NOT NULL,
    locationCity VARCHAR(30) NOT NULL,
    locationState VARCHAR(25) NOT NULL,
    locationZipCode VARCHAR(15) NOT NULL,
    locationPhoneNumber VARCHAR(15) NOT NULL,
    locationLatitude VARCHAR(255) NOT NULL,
    locationLongitude VARCHAR(255) NOT NULL,
    locationCategory VARCHAR(255) NOT NULL,
    locationImage VARCHAR(255) NOT NULL,
    neighborhood VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    locationRating TINYINT(1) NOT NULL,
    active TINYINT(1) NOT NULL,
    deleted TINYINT(1) NOT NULL
);

CREATE TABLE tblUsers (
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    username VARCHAR(255) NOT NULL,
    profileImage VARCHAR(255) NOT NULL,
    timeSignedUp DATETIME NOT NULL,
    locationNumber VARCHAR(255),
    locationRoad VARCHAR(255),
    locationNeighborhood VARCHAR(255),
    locationCity VARCHAR(255),
    locationCounty VARCHAR(255),
    locationState VARCHAR(255),
    locationZipCode VARCHAR(255),
    locationCountry VARCHAR(255),
    locationCountryCode VARCHAR(255),
    locationLat VARCHAR(255),
    locationLon VARCHAR(255),
    locationSet TINYINT(1),
    emlMarketing TINYINT(1) NOT NULL,
    deleted TINYINT(1) NOT NULL,
    verified TINYINT(1) NOT NULL,
    banned TINYINT(1) NOT NULL,
    isAdmin TINYINT(1) NOT NULL,  
);
